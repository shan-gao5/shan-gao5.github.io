\section{Continuation of Hausdorff and etc}
\begin{proposition}[Limits of sequeneces in Hausdorff space are unique]
    IF $X$ is Hausdorff, then a sequence in $X$ has at most one limit
    
\end{proposition}
\begin{proof}
    Suppose $\{x_n\}_{n \in \mathbb{N}}$ converges to $x$ and $y$ with $x \neq y$. By $T_2$, we have that there exists $U$ and $V$ open in $X$ such that $x \in U$ and $y \in V$ and $X \cap V = \emptyset$. Then that means that there exists $N_1$ such that if $n \geq N_1$, then $x_n \in U$, similarly $\exists N_2$ such that if $n \geq N_2$, $x_n \in V$. Then pick $N = \max(N_1, N_2)$, we have that $x_n \in U$ and $x_n \in V$, which contradicts the fact that $U \cap V = \emptyset$. 
\end{proof}
\begin{definition}[basis]
    let $X$ be a topological space and $\mathcal{B}$ is called a basis if it's a collection of sets such that for all $U \in X$ open in $X$, $U$ is a union of elements in $\mathcal{B}$, so 
    \[
        U = \bigcup_{B \subseteq U \colon B \in \mathcal{B}} B
        \]
	In other words, each open set $U$ is the union of all basic open sets in $U$

\end{definition}
\begin{definition}[axioms of basis collection]
	A collection of subsets of $X$ is called a basis if 
	\begin{itemize}
		\item $\forall x \in X$, $\exists B \in \mathcal{B}$ such that $x \in B$
		\item $\forall B_1, B_2 \in \mathcal{B}$ and $x \in B_1 \cap B_2$, $\exists B_3 \in B_1 \cap B_2$ and $x \in B_3$
	\end{itemize}
\end{definition}
\begin{lemma}[defining topology off basis]
	How to define a topology on $X$ whose open sets are unions of elements of $\mathcal{B}$ TFAE
	\begin{itemize}
		\item $U \in X$ is open $\iff$ it's the union of basic open sets contained in $U$, $$ U = \bigcup_{B \in U\colon B \in \mathcal{B}} B$$
		\item $U \in X$ is open $\iff \forall x \in U$, there exists a $B \in \mathcal{B}$ such that $x \in B$ and $B \subseteq U$. 
	\end{itemize}
\end{lemma}
\begin{proof}
	\begin{itemize}
		\item Union of every element of $\mathcal{B}$ in $X$ is $X$ so $X$ is open, and $\emptyset$ is the empty union so it's open as well
		\item (Finite intersection) let $x \in U \cap V$ where $U \cap V$ are open sets as described previously using basic sets, then $\exists B_1 \in U$ and $\exists B_2 \in V$ such that $B_1, B_2 \in \mathcal{B}$. This means that there exists a $B_3 \in \mathcal{B}$ such that $B_3 \subseteq B_1 \cap B_2$, then by the second axiom for basis, $x \in B_3$, which, by our lemma, along with the fact that $B_3 \subseteq U \cap V$, $U \cap V$ is open.
	\end{itemize}
	There are many ways to define a topology on a product space.
	\begin{definition}[Product Topology]
		$X$ and $Y$ are topological spaces, $X \times Y = \{(x,y) \colon x \in X, y \in Y\}$ is a topological space if we endow it with the following definition of open sets : $$\text{ open sets } = \{ U \times V \colon U \subseteq X \text{ open }, V \subseteq Y \text { open } \}$$	
	\end{definition}
	\begin{proof}
		This is a basis, we just need to verify the axioms for a basis. Let $(x,y) \in X \times Y$, then obviously since $x \in X$ and $y \in Y$, there exists $U \subseteq X$ and $V \subseteq Y$ such that $x \in U$ and $y \in V$, thus $(x,y) \in U \times V$. For the second axiom, let $U_1 \times V_1$ and $U_2 \times V_2$ be two basis elements and $(x,y) \in (U_1 \times V_1 ) \cap (U_2 \times V_2) = (U_1 \cap U_2) \times (V_1 \cap V_2)$, thus $x \in (U_1 \cap U_2)$ and $y \in V_1 \cap V_2$, therefore there exists so letting $B_3 = (U_1 \cap U_2) \times ( V_1 \cap V_2)$ be a basic open set (finite intersection), we are done. 

	\end{proof}

		\begin{itemize}
			\item $f(z) = p(h(z))$
			\item $g(z) = q(h(z))$
			\item $h(z) = (f(z), g(z))$

		\end{itemize}
	\end{definition}
