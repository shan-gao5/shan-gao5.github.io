\begin{definition}[algebra]
	Let $S$ be a set, an \textbf{algebra} on a set of subsets of $S$: $\Sigma_0$ is called an algebra on $S$ (or an algebra on $2^S$) if 
	\begin{enumerate}
		\item $S \in \Sigma_0$ 
	\item $F \in \Sigma_0 \implies F^c := S\setminus F \in \Sigma_0$ 
	\item $F, G \in \Sigma_0 \implies F \cup G \in \Sigma_0$ 
		
	\end{enumerate}
	From these rules, obvious other rules follows such as, $\emptyset \in \Sigma_0$ obviously since $\emptyset = S^c \in \Sigma_0$. Also letting $F, G \in \Sigma_0$, then $(F^c \cup G^c)^c = F\cap G \in \Sigma_0$ 
	Thus, an algebra on S is a family of subsets of $S$ that are stable under finitely many set operations.
\end{definition}
This definition is algebraic in the sense that $\cap$ is the product operation and symmetric difference being 
\begin{align*}
	A \Delta B := (A \cup B)\setminus(A \cap B)
\end{align*}
We can extend this definition to form a $\sigma$-algebra on $S$ which takes the rules and extends them towards their countable definition. That means, if $\Sigma$ is an algebra on $S$, such that whenever $F_n \in \Sigma$ ($n \in \mathbb{N}$) then 
\begin{align*}
	\bigcup_{n} F_n \in \Sigma
\end{align*}
Obviously set intersection follows
\begin{align*}
	\bigcap_n F_n = \left( \bigcup_n F_n^c \right)^c \in \Sigma
\end{align*}
\subsection{measurable space}
A pair $(S, \Sigma)$, where $S$ is a set and $\Sigma$ is a $\sigma$-algebra on $S$, is called a measurable space. An element of $\Sigma$ is called a $\Sigma$-measurable subset of $S$. 
\begin{definition}[$\sigma$-algebra generated by a class $\mathcal{C}$ of subsets]
	Let $\mathcal{C}$ be a class of subsets of $S$, then $\sigma(\mathcal{C})$, the $\sigma$-algebra generated by $\mathcal{C}$ is the smallest $\sigma$-algebra $\Sigma$ on $S$ such that $\mathcal{C} \subseteq \Sigma$. 
	
\end{definition}
\begin{proposition}
	A simple proposition follows 
	\begin{itemize}
		\item if C is a $\sigma$-algebra then $\sigma(\mathcal{C}) = \mathcal{C}$ 
		\item $\sigma(\sigma(\mathcal{C})) = \sigma(\mathcal{C})$
		\item If $C_1 \subseteq C_2$, then $\sigma(C_1) \subseteq \sigma(C_2)$
	\end{itemize}
\end{proposition}
By construction the smallest sigma algebra is the intersection of all sigma algebras that contain $\mathcal{C}$. An interesting fact that arises from this is that the intersection of any two sigma algebra remains a sigma algebra. 
\begin{example}
	An example would be if $A, B \subseteq S$, what is $\sigma(\{A\})$. We just need to add everything that makes it a $\sigma$-algebra, that means closure under union and intersection and complements and the set $S$ itself, that means $\sigma(\{A\}) = \{ A, A^c = B	, \emptyset, S\}$
\end{example}
\begin{definition}[borel $\sigma$-algebra]
	

An easy example would be a borel $\sigma$-algebra, $\mathcal{B}(S)$. Indeed, letting $S$ be a topological space, $B(S)$ is the borel $\sigma$-algebra generated by the family of open subsets of $S$. $\mathcal{B}(\mathbb{R})$ is the most important of all $\sigma$-algebras, oftenhand shorthanded by $\mathcal{B}$.
\begin{align*}
	\mathcal{B}(\mathbb{R}) = \sigma(\text{open subsets of $\mathbb{R}$})
\end{align*}
\end{definition}
Consider the collection 
\begin{align*}
	\pi(\mathbb{R}) := \{ (-\infty, x] \colon x \in \mathbb{R}\}
\end{align*}
We claim that \[\mathcal{B} = \sigma(\pi(\mathbb{R})\]
\begin{proof}
	\begin{itemize}
		\item ($\supset$) Since they are both $\sigma$-algebras, we just need to prove that they contain the same elements. We can do this exhaustively.  Taking an element of $\sigma(\pi(\mathbb{R}))$ as $(-\infty, x)$ for some $x$, we notice that this is an element of $\mathcal{B}$, since $(-\infty, x] =\bigcap_{n \in \mathbb{N}} \left(-\infty, x+\frac{1}{n}\right)$
		\item $\subset$ all that remains to be prove that is every open set $G$ of $\mathcal{B}$ is in $\sigma(\pi(\mathbb{R}))$, we recall from first problem set of analysis 2 that every open set $G$ is a countable union of open intervals, that means we just need to show that any $(a,b)$ for $a,b \in \mathbb{R}$ with $a< b$ is in $\sigma(\pi(\mathbb{R}))$. But for any $u$ with $u > a$, we have that 
			\begin{align*}
				(a,u] =(-\infty,u] \cap (a,+\infty) = (-\infty,u] \cap (-\infty, a]^c \in \sigma(\pi(\mathbb{R}))
			\end{align*}

			but for $\varepsilon = \frac{1}{2}(b-a)$, we get that 
			\begin{align*}
				(a,b) = \bigcup_n (a,b-\varepsilon \frac{1}{n}]
			\end{align*}
			to recover the open interval $(a,b)$ in $\mathbb{R}$

	\end{itemize}	
\end{proof}

\begin{definition}[set theoretic limits]
"$A_n \uparrow A$" means that 
\begin{enumerate}
	\item it's Monotonically increasing $A_n \subseteq A_{n+1}$
	\item $\bigcup_n^\infty A_n = A$
\end{enumerate}
"$A_n \downarrow A$ means that 
\begin{enumerate}
	\item Monotonically decreasing $A_n \supseteq A_{n+1}$ 
	\item $\bigcap_n^\infty A_n = A$
\end{enumerate}
	
\end{definition}
	\begin{definition}[d-systems]
			A collection $\mathcal{D}$ of subsets of $S$ is called a d-system if 
			\begin{enumerate}
				\item $S \in \mathcal{D}$ 
				\item Closed under set differences $A, B \in \mathcal{D}$, then $A \setminus D \in \mathcal{D}$ 
				\item Closed under monotonic limit $(A_n) \subseteq \mathcal{D}$ if $A_n \uparrow A$ then $A \in \mathcal{D}$ 
			\end{enumerate}
	\end{definition}	
	\begin{lemma}
		let $\Sigma$ be a collection of subsets of $S$, then 
		\begin{align*}
			\Sigma \text{ is a $\sigma$-algebra } \iff \Sigma \text{ is a d-system and a $\pi$-system } 
		\end{align*}
		check definition \ref{def:pi-system} for $\pi$-system
	\end{lemma}

	\begin{definition}[set functions]
		Let $S$ be a set, letting $\Sigma_0$ be an algebra, and let $\mu_0$ be a non-negative set function 
		
	\end{definition}
	\begin{definition}[Additive]
		$\mu_0$ is called additive if $\mu_0 (\emptyset) = 0$ and, for $F,G \in \Sigma_0$
		\begin{align*}
			F\cap G = \emptyset \implies \mu_0(F \cup G) = \mu_0(F) + \mu_0(G)
		\end{align*}
	\end{definition}
It is countable additive if letting $(A_n) \subseteq \Sigma_0$ and $A_i \cap A_j = \emptyset$ if $i \neq j$, then 
$$
	\mu_0 \left( \bigcup_i A_i \right) = \sum_i \mu_0(A_i)
$$
\begin{definition}[measure space]
	letting $\Sigma_0$ be a $\sigma$-algebra on $S$, from the measurable space $(S,\Sigma_0)$, we can form a measure space if we have a map $\mu \colon \Sigma_0 \to [0,\infty)$ that is countably additive then the triple $(S, \Sigma_0, \mu)$ is called a \textbf{measure space}
\end{definition}
A measure space is \textbf{finite} if 
$$
	\mu(S) < \infty
$$
\\
$\sigma$-finite if 
$$
\exists (S_n \colon n \in \mathbb{N}) 	\colon \mu(S_n) < \infty (\forall n \in \mathbb{N}) \text{ and } \bigcup S_n = S
$$

Finally, we can introduce the probability measure: \\
A measure $\mu$ is called a \textit{probability measure} if 
$$
	\mu(S) = 1
$$
then the measure space $(S, \Sigma, \mu)$ is called a probability \textbf{triple}

An element $F \in S$ is called $\mu$-null if $\mu(F) = 0$. A statement about of set of points $s \in S$ is said to hold \textit{almost everywhere} if, letting the set of points $s$ be $F$, so
$$
	F:= \{s \in S: \text{statement} \} \in \Sigma 
$$
almost everwhere if $\mu(F) = 0$\\

\begin{proposition}[some basic properties of measures]
	We aim to introduce some basic results
	\begin{enumerate}
		\item \textbf{monotonicity} if $A,B \in \Sigma$, and $A \subseteq B$, then $\mu(A) \leq \mu(B)$
		\item \textbf{subadditive} Letting $(A_n) \subseteq \Sigma$, then  
		\item 
		$$
			\mu \left(\bigcup_i^\infty A_i\right) \leq \sum_i^\infty \mu(A_i)
		$$ Note that subadditivity is applicable to the finite case as well.
		
	\end{enumerate}
	
	
\end{proposition}

\begin{proposition}[Continuity 1]
Letting $A_n \in \Sigma$ for all $n$. If $A_n \uparrow A$, then 
\begin{align*}
	\mu\left(\bigcup_n^\infty A_n\right) = \lim_{n \to \infty} \mu(A_n)
\end{align*}
	\begin{proof}
		Letting $B_1 = A_1$, and setting $B_n = A_{n+1}\setminus A_n$, notice that all the $B_i$'s are disjoint (draw a picture, knowing that $A_n \subseteq A_{n+1}$ for all $n$). We know that, by construction, 
		\[
			\bigcup_n^\infty B_n = \bigcup_n^\infty A_n
		\]
		Then by countable additivity, we have 
		\begin{align*}
			\mu(\bigcup_i B_i) = \sum_i \mu(B_i) &= \lim_{n \to \infty} \sum_i^n \mu(B_i)\\
			&= \lim_{n \to \infty} \mu\left(\bigcup_i^n B_i\right)\\
			&= \lim_{n \to \infty} \mu(A_i) 
		\end{align*}
	\end{proof}
\end{proposition}
\begin{proposition}[continuity 2]
If $A_n \downarrow A$, then 
\[
	\lim_{n \to \infty} \mu(A_n) = \mu\left(\bigcap_n A_n\right)
\]
\end{proposition}

\begin{theorem}
Given a set $S$ and an algebra $\Sigma_0$, assume that $\mu$ is a non-negative set function:
$$\mu \colon \Sigma_0 \to [0,\infty[$$
(that means finitely additive), then \\
$\mu$ is \textbf{countably additive} $\iff (A_n) \subseteq \Sigma_0 $ and $A_n \downarrow \emptyset$ then $\lim_{n \to \infty} \mu(A_n) =0$. Remember that since $\mu$ is a additive set function, then $\mu(\emptyset) = 0$. THIS IS THE SEQUENTIAL REPRESENTATION OF CONTINUITY at a point in the domain $\Sigma_0$. 
\end{theorem}
This is straightforward to prove by unraveling the definition of each statement.

\begin{definition}[$\pi$-systems] \label{def:pi-system}
	Let $S$ be a set, a $\pi$-system on $S$ is a family of subsets $\mathcal{I}$ on $S$ that is stable under finite intersection, that means letting $I_1, I_2 \in I$, then 
	$$
	I_1 \cap I_2  \in \mathcal{I}
	$$

\end{definition}
Now if we generated a $\sigma$-algebra on $\mathcal{I}$, that means $\Sigma = \sigma(\mathcal{I}))$, and we have two measures $\mu_1$ and $\mu_2$ on $(S,\Sigma)$, such that $\mu_1 = \mu_2$ on $\mathcal{I}$ and they are both finite, then $\mu_1 = \mu_2$ on $\Sigma$. (Recall that generating a $\sigma$-algebra means taking the intersection of all $\sigma$-algebras that have $\mathcal{I}$ as a subclass.
\begin{corollary}
	A corollary follows immediately, if two measures agree on a $\pi$-system, then they agree on the $\sigma$-algebra generated from the $\pi$-system.
\end{corollary}
\begin{theorem}[Carathéodory's Extension Theorem]
	Let S be a set, let $\Sigma_0$ be an algebra on $S$, and let 
	$$
		\Sigma := \sigma(\Sigma_0)
	$$
	Then if $\mu \colon \Sigma_0 \to [0,+\infty]$ is a countably additive map then there exists a mesure $\mu$ on $(S, \Sigma)$ such that
	$$
		u = \mu_0 \text{ on } \Sigma_0
	$$
\end{theorem}
By corollary 1.6.1 that means if $\mu_0$ is finite, then the induced measure is \textbf{unique}

\begin{definition}[Lebesgue measure]
	Let $S = (0,1]$. For $F \subseteq S$, say that $F \in \Sigma_0$ if $F$ may be written as a finite union 
	$$
		(a_1,b_1] \cup \ldots \cup (a_r, b_r]
	$$
	where $r \in \mathbb{N}$, and $0 \leq a_1 \leq b_1 \leq \ldots \leq a_r \leq b_r \leq $. Then $\Sigma_0$ is an algebra on $(0,1]$ and
	$$
		\Sigma := \sigma(\Sigma_0) = \mathcal{B}(0,1]
	$$
	For $F$ as defined previously, letting 
	$$
		\mu_0(F) = \Sigma_{k \leq r}(b_k -a_k)
	$$
	Then $\mu_0$ is well defined and additive on $\Sigma_0$. Moreover $\mu_0$ is countably additive on $\Sigma_0$. Hence by the previous theorem, there exists a unique measure $\mu$ on $(0,1]$ extending $\mu_0$ on $\Sigma_0$. The lesbesgue measure on $[0,1]$ is obtained by a trivial modification setting the set $\{0\}$ to have lebesgue measure 0.
\end{definition}
\begin{lemma}[Elementary inequalities]
	Let $(S,\Sigma,\mu)$ be a measured space. Then 
	\begin{enumerate}
		\item $\mu(A\cup B) \leq \mu(A) + \mu(B) (A,B \in \Sigma)$
		\item $\mu\left(\bigcup_{i \leq n} F_i\right) \leq \sum_{i \leq n} \mu(F_i)$
		\item if $\mu(S) < \infty$, then 
			$$
				\mu(A \cup B) = \mu(A) + \mu(B) - \mu(A \cap B) \quad (A,B \in \Sigma)
			$$
	\end{enumerate}

\end{lemma}
